# When - Kolin doesn't have Switch, instead uses "When" 
- Example
```
fun main() {
    val number = 2

    when (number) {
        1 -> println("One")
        2 -> println("Two")
        3 -> println("Three")
        else -> println("Unknown number")
    }
}
```
```
fun main() {
    val name = "Sarah"

    when (name) {
        "Tony" -> println("T")
        "john" -> println("J")
        "Sarah" -> println("S")
        else -> println("Unknown name")
    }
}
```
